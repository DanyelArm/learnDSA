    ```typescript
    import { xpForLevel } from '@dsa-quest/shared'

    // Generate thresholds dynamically from the shared formula
    // Level N requires xpForLevel(N) cumulative XP
    export function getLevelProgress(xp: number, serverLevel: number): { progress: number; currentThreshold: number; nextThreshold: number } {
      const currentThreshold = xpForLevel(serverLevel)
      const nextThreshold = xpForLevel(serverLevel + 1)
      const progress = nextThreshold > currentThreshold
        ? (xp - currentThreshold) / (nextThreshold - currentThreshold)
        : 1
      return { progress: Math.max(0, Math.min(progress, 1)), currentThreshold, nextThreshold }
    }
    ```

    ```typescript
    import type { AchievementDTO } from '@dsa-quest/shared'

    export type ModalItem =
      | { type: 'level-up'; newLevel: number }
      | { type: 'badge'; achievement: AchievementDTO }

    interface GamificationState {
      modalQueue: ModalItem[]
      enqueue: (items: ModalItem[]) => void
      dequeue: () => void
      clearQueue: () => void
    }

    export const useGamificationStore = create<GamificationState>()((set) => ({
      modalQueue: [],
      enqueue: (items) => set((s) => ({ modalQueue: [...s.modalQueue, ...items] })),
      dequeue: () => set((s) => ({ modalQueue: s.modalQueue.slice(1) })),
      clearQueue: () => set({ modalQueue: [] }),
    }))
    ```

    **5. Create `processGamificationResponse` utility in `client/src/lib/gamificationHelper.ts`:**

    Create a dedicated file for the helper function that takes a GamificationResult from any API response and:
    1. Calls `authStore.getState().updateUser({ xp: result.user.xp, level: result.user.level })`
    2. Builds modal items: if `result.leveledUp`, push `{ type: 'level-up', newLevel: result.user.level }` FIRST (per D-23)
    3. For each achievement in `result.unlockedAchievements`, push `{ type: 'badge', achievement }`
    4. Calls `gamificationStore.getState().enqueue(items)`

    ```typescript
    import type { GamificationResult } from '@dsa-quest/shared'
    import { useAuthStore } from '@/stores/authStore'
    import { useGamificationStore, type ModalItem } from '@/stores/gamificationStore'

    export function processGamificationResponse(result: GamificationResult): void {
      // Update user XP/level reactively
      useAuthStore.getState().updateUser({ xp: result.user.xp, level: result.user.level })

      // Build modal queue: level-up FIRST, then badges (per D-23)
      const items: ModalItem[] = []
      if (result.leveledUp) {
        items.push({ type: 'level-up', newLevel: result.user.level })
      }
      for (const achievement of result.unlockedAchievements) {
        items.push({ type: 'badge', achievement })
      }
      if (items.length > 0) {
        useGamificationStore.getState().enqueue(items)
      }
    }
    ```
