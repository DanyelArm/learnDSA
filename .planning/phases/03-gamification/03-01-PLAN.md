    **4. Create `server/src/lib/xpCalculation.ts`:**

    ```typescript
    import { XP_PER_STAGE, XP_BONUS_FIRST_TRY, XP_BONUS_SPEED, xpForLevel } from '@dsa-quest/shared'

    export type StageType = 'theory' | 'practice' | 'challenge'

    export interface XpAwardInput {
      attempts: number
      elapsedSeconds: number | null
      timeBonusThreshold: number | null
    }

    export interface XpAwardResult {
      base: number
      firstTryBonus: number
      speedBonus: number
      total: number
    }

    export function calculateXpAward(stage: StageType, input: XpAwardInput): XpAwardResult {
      const base = XP_PER_STAGE[stage]
      const firstTryBonus = input.attempts === 1 ? XP_BONUS_FIRST_TRY : 0
      const speedBonus = (input.elapsedSeconds !== null && input.timeBonusThreshold !== null && input.elapsedSeconds <= input.timeBonusThreshold)
        ? XP_BONUS_SPEED : 0
      return { base, firstTryBonus, speedBonus, total: base + firstTryBonus + speedBonus }
    }

    export function deriveLevel(totalXp: number): number {
      let level = 1
      while (xpForLevel(level + 1) <= totalXp) level++
      return level
    }
    ```
