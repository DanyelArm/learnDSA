    **2. Update `GamificationOrchestrator.tsx`:**

    Replace the placeholder modal rendering with the actual styled components:
    ```tsx
    import { LevelUpModal } from './LevelUpModal'
    import { BadgeModal } from './BadgeModal'

    export function GamificationOrchestrator() {
      const { modalQueue, dequeue } = useGamificationStore()
      const current = modalQueue[0] ?? null

      return (
        <AnimatePresence mode="wait">
          {current?.type === 'level-up' && (
            <LevelUpModal
              key="level-up"
              newLevel={current.newLevel}
              onDismiss={dequeue}
            />
          )}
          {current?.type === 'badge' && (
            <BadgeModal
              key={`badge-${current.achievement.key}`}
              achievement={current.achievement}
              onDismiss={dequeue}
            />
          )}
        </AnimatePresence>
      )
    }
    ```

    Key change: add `mode="wait"` to `AnimatePresence` so exit animation completes before the next modal enters. This ensures sequential display per D-23.
